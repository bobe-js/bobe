```typescript
interface RouterOptions {
  routes?: RouteMap;
  initialPath?: string;
  scrollRootId?: string;
}
```

```typescript
class Router {
  /** 创建 Router 实例，读取构造参数、插件注入数据、SSR path 或浏览器 location 初始化首屏路由 */
  constructor(opt?: RouterOptions);

  /** 当前激活路由；模板通过 active.component / active.layout 渲染当前页面 */
  active: RouteEntry | null;

  /** 路由表；保存 path pattern 到 RouteRecord 的映射 */
  routes: RouteMap;

  /** 菜单树；由插件扫描页面目录生成，供布局和导航组件使用 */
  menus: Menu[];

  /** 滚动容器 DOM id；hash 滚动和历史滚动恢复优先作用到该元素 */
  scrollRootId?: string;

  /** 最大并行预加载组件数量 */
  maxPreload: number;

  /** 进入守卫；导航提交前检查目标路由是否允许进入 */
  enterGuard?: (to: RouteEntry) => GuardResult | Promise<GuardResult>;

  /** 离开守卫；导航提交前检查当前路由是否允许离开 */
  leaveGuard?: (from: RouteEntry) => GuardResult | Promise<GuardResult>;

  /** 等待首屏路由初始化完成；无参数时返回 Promise */
  ready(): Promise<void>;

  /** 等待首屏路由初始化完成；传入回调时在 ready 后执行 */
  ready(cb: () => void): void;

  /** 追加一条浏览器历史并导航到 url */
  pushState(url: string): Promise<void>;

  /** 替换当前浏览器历史并导航到 url */
  replaceState(url: string): Promise<void>;

  /** 后退一条浏览器历史 */
  back(): Promise<void>;

  /** 前进一条浏览器历史 */
  forward(): Promise<void>;

  /** 按 delta 跳转多条浏览器历史 */
  go(delta: number): Promise<void>;

  /** 处理 document 点击事件；将可拦截的同源 a 标签点击转成 Router 导航 */
  onClick(e: MouseEvent): void;

  /** 处理浏览器 popstate；同步浏览器历史变化、active、hash 滚动和滚动恢复 */
  onPopstate(e: PopStateEvent): void;

  /** 处理 hash 滚动；通常在 active 渲染完成后的 post effect 中执行 */
  onHashScroll(): void;

  /** 处理空闲预加载；在浏览器空闲时预加载还未访问的路由组件 */
  onIdlePreload(): void;

  /** 处理路由组件异步加载；调用 route.import 并同步 component / meta / layout */
  onRouteImport(path: string): Promise<unknown>;

  /** 处理滚动恢复；根据当前历史项恢复 scrollRoot 或 window 的滚动位置 */
  onScrollRestore(): void;
}
```
