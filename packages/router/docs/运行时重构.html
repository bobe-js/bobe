Router 运行时重构流程

这份文档按 Router 当前 runtime 实现整理对外 API 和运行时 hook。导航通过同一条事务管线提交， 守卫只执行一次，浏览器历史用 router 自有 state 识别，滚动在 active 渲染完成后统一处理。

导航事务：match、build target、guard、load、commit、scroll、preload 必须按一次事务提交。
守卫只跑一次：public history API 和 popstate 之间用 pending token 衔接，不重复派发。
history.state：用 historyKey / entryId / index 识别历史项，避免重复 URL 或刷新后内存栈重置导致 stackIndex 错乱。
失败回滚：guard 拒绝或 import 失败时，不提交 active；必要时补偿浏览器历史。
事务执行器只负责 handler 串行和 token 校验；token 失效默认抛弃结果，不做通用撤回。
首屏守卫只能来自 RuntimeRouterOptions；构造后再赋值的 guard 只能影响后续导航。
滚动作为 handleScroll 留在事务队列中；active 提交前创建一次性 RenderWaiter，handleScroll 通过 Promise + await 等渲染完成后执行。
异步 handler 在 await 后、产生副作用前必须校验 token；事务执行器只负责 handler 边界的兜底校验。

#runNavigation(request, handlers)

进入 hook/api 后的流程图

flowchart TD
  A["Router 内部调用: #runNavigation(request, handlers)"] --> A1["NavigationTransactionRunner.run(router, handlers, request)"]
  A1 --> B["创建 NavigationContext: router / request / token / status=running / rollbackStack=[]"]
  B --> C["createToken(request): token.id++ 并标记为 current"]
  C --> D["复制 handler queue = [...handlers]"]
  D --> E{"queue 是否为空?"}
  E -- "yes" --> F["若 status 仍为 running, 设置 completed 并返回 NavigationResult"]
  E -- "no" --> G["shift 下一个 NavigationHandler"]
  G --> H["await handler(ctx)"]
  H --> I{"handler 是否抛错?"}
  I -- "yes" --> J["ctx.error = error; ctx.status = error"]
  J --> K["倒序执行 ctx.rollbackStack"]
  K --> L["返回 NavigationResult(status=error, ctx, error)"]
  I -- "no" --> M{"token 是否仍为 current 且未 cancelled?"}
  M -- "no" --> N["ctx.status = cancelled; 返回 cancelled"]
  M -- "yes" --> O{"handler result 类型?"}
  O -- "void" --> D
  O -- "stop" --> P["ctx.status = result.status ?? ctx.status; 返回"]
  O -- "prepend" --> Q["queue.unshift(...result.handlers)"]
  Q --> D
  O -- "replace" --> R["queue = [...result.handlers]"]
  R --> D
  L --> S["#runNavigation 读取 NavigationResult(error)"]
  S --> T{"router.onError 是否存在?"}
  T -- "yes" --> U["await onError(error, ctx); 用户自行弹窗或记录"]
  T -- "no" --> V["console.error(error)"]
  U --> W["返回 NavigationResult; 不切 errorComponent"]
  V --> W

统一数据类型

RuntimeRouterOptions

interface RuntimeRouterOptions {
  routes?: RouteMap;
  initialPath?: string;
  enterGuard?: (to: RouteEntry) => GuardResult | Promise<GuardResult>;
  leaveGuard?: (from: RouteEntry) => GuardResult | Promise<GuardResult>;
  onError?: RouterErrorHandler;
  loadTimeout?: number;
  onTimeout?: RouterTimeoutHandler;
  scrollRootId?: string;
  maxPreload?: number;
  historyKey?: string;
}

RouteMap

type RouteMap = Record<string, RouteRecord>;

RouteRecord

type RouteRecord = {
  import?: () => Promise<RouteModule>;
  component?: typeof Store;
  status: 'idle' | 'loading' | 'loaded' | 'error';
  promise?: Promise<unknown>;
  params?: Record<string, string>;
  meta?: Record<string, unknown>;
  layout?: typeof Store | (() => Promise<RouteModule>);
};

RouteEntry

type RouteEntry = {
  id: string;
  path: string;
  url: string;
  hash: string;
  params: Record<string, string>;
  scroll?: ScrollSnapshot;
  component?: typeof Store | Store;
  meta?: Record<string, unknown>;
  layout?: typeof Store | (() => Promise<RouteModule>);
};

Menu

type Menu = {
  name: string;
  path?: string;
  hasComponent: boolean;
  children?: Menu[];
  meta?: Record<string, unknown>;
  nearestFile?: {
    name: string;
    path: string;
  };
};

RouteMatch

type RouteMatch = {
  pattern: string;
  path: string;
  url: string;
  search: string;
  hash: string;
  params: Record<string, string>;
  record: RouteRecord;
};

GuardResult

type GuardResult =
  | boolean
  | {
      ok: boolean;
      redirect?: string;
    };

NavigationRequest

type NavigationRequest =
  | {
      source: 'init' | 'api' | 'click' | 'popstate' | 'external-popstate' | 'redirect';
      url: string;
      historyMode: 'push' | 'replace' | 'none';
      runGuards: boolean;
      restoreScroll: boolean;
    }
  | {
      source: 'api';
      historyMode: 'browser-delta';
      delta: number;
      runGuards: boolean;
      restoreScroll: true;
    };

NavigationContext

type NavigationContext = {
  router: RuntimeRouterLike;
  request?: NavigationRequest;
  token: NavigationToken;
  status: NavigationStatus;
  url?: string;
  match?: RouteMatch;
  from?: RouteEntry | null;
  to?: RouteEntry;
  historyDelta?: HistoryDeltaRequest;
  pendingHistoryDelta?: PendingHistoryDelta;
  guard?: GuardDecision;
  commitPlan?: NavigationCommitPlan;
  scrollIntent?: ScrollIntent;
  renderWaiter?: RenderWaiter;
  error?: unknown;
  rollbackStack: RollbackHandler[];
};

NavigationHandler

type NavigationHandler =
  (ctx: NavigationContext) =>
    NavigationHandlerResult | Promise<NavigationHandlerResult>;

NavigationHandlerResult

type NavigationHandlerResult =
  | void
  | {
      type: 'stop';
      status?: NavigationStatus;
    }
  | {
      type: 'prepend';
      handlers: NavigationHandler[];
    }
  | {
      type: 'replace';
      handlers: NavigationHandler[];
    };

NavigationToken

type NavigationToken = {
  id: number;
  request?: NavigationRequest;
  cancelled: boolean;
};

NavigationStatus

type NavigationStatus = 'running' | 'completed' | 'ignored' | 'blocked' | 'cancelled' | 'error';

NavigationResult

type NavigationResult = {
  status: NavigationStatus;
  ctx: NavigationContext;
  error?: unknown;
};

RouterErrorHandler

type RouterErrorHandler =
  (error: unknown, ctx: NavigationContext) => void | Promise<void>;

RouterTimeoutDecision

type RouterTimeoutDecision = 'continue' | 'cancel';

RouterTimeoutHandler

type RouterTimeoutHandler =
  (ctx: NavigationContext) =>
    RouterTimeoutDecision | Promise<RouterTimeoutDecision>;

RollbackHandler

type RollbackHandler =
  (ctx: NavigationContext) => void | Promise<void>;

RuntimeRouterLike

interface RuntimeRouterLike {
  active: RouteEntry | null;
  routes: RouteMap;
  menus: Menu[];
  enterGuard?: (to: RouteEntry) => GuardResult | Promise<GuardResult>;
  leaveGuard?: (from: RouteEntry) => GuardResult | Promise<GuardResult>;
  scrollRootId?: string;
}

NavigationCommitPlan

type NavigationCommitPlan = {
  historyState?: RouterHistoryState;
  historyAction: 'push' | 'replace' | 'none';
  stackAction: 'append' | 'replace-current' | 'move-index' | 'reset' | 'none';
  nextStackIndex?: number;
  scrollIntent?: ScrollIntent;
  renderWaiter?: RenderWaiter;
};

GuardDecision

type GuardDecision =
  | { type: 'allowed' }
  | { type: 'blocked'; result: GuardResult }
  | { type: 'redirect'; to: string; result: GuardResult };

RouterHistoryState

type RouterHistoryState = {
  __bobeRouter: true;
  historyKey: string;
  entryId: string;
  index: number;
  url: string;
};

HistoryDeltaRequest

type HistoryDeltaRequest = {
  delta: number;
  targetState?: RouterHistoryState;
  targetEntry?: RouteEntry;
};

PendingHistoryDelta

type PendingHistoryDelta = {
  tokenId: number;
  delta: number;
  fromIndex: number;
  toIndex: number;
  targetEntryId: string;
  targetUrl: string;
  consumed: boolean;
};

ScrollIntent

type ScrollIntent =
  | {
      type: 'hash';
      hash: string;
      retry: ScrollRetryPolicy;
      behavior?: ScrollBehavior;
    }
  | {
      type: 'restore';
      top: number;
      left: number;
      retry: ScrollRetryPolicy;
    }
  | {
      type: 'top';
      retry: ScrollRetryPolicy;
    };

ScrollRetryPolicy

type ScrollRetryPolicy = {
  attempts: number;
  maxAttempts: number;
  wait: 'post-effect' | 'raf';
};

RenderWaiter

type RenderWaiter = {
  /** 在 active 提交前创建，避免错过由 active 触发的 post effect */
  promise: Promise<void>;
  source: 'post-effect' | 'raf' | 'resolved';
  dispose?: () => void;
};

ScrollSnapshot

type ScrollSnapshot = {
  top: number;
  left: number;
};

ScrollTarget

type ScrollTarget = {
  root: HTMLElement | Window;
  scrollRootId?: string;
};

PreloadTask

type PreloadTask = {
  path: string;
  scheduledAt: number;
  fromIdleQueue: true;
};

RouteModule

type RouteModule =
  | typeof Store
  | {
      default?: typeof Store;
      routeMeta?: Record<string, unknown>;
      layout?: typeof Store | (() => Promise<RouteModule>);
    };

ReadyCallback

type ReadyCallback = () => void;
